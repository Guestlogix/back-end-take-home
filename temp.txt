FROM microsoft/dotnet:2.1-sdk AS build-env
WORKDIR /build
COPY . /build

#Restore
RUN dotnet restore ./airtrip.sln

RUN dotnet publish ./src/AirTrip.Main/ -o /publish --configuration Release

# Build runtime image
FROM microsoft/dotnet:2.1-runtime
WORKDIR /app
COPY --from=build-env /publish/ .

ENTRYPOINT ["dotnet", "AirTrip.dll"]
